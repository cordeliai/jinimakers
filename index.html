function downloadVCard() { const vcard = "BEGIN:VCARD\n" + "VERSION:3.0\n" + "FN:김유진 (코드스튜디오)\n" + "TEL;TYPE=CELL:01065581165\n" + "EMAIL:jinimakers@kakao.com\n" + "ORG:지니메이커스;코드스튜디오\n" + "URL:" + window.location.href + "\n" + "END:VCARD"; // 💡 핵심: UTF-8 BOM(0xEF, 0xBB, 0xBF)을 추가하여 안드로이드/아이폰에서 한글 인식률을 극대화합니다. const blob = new Blob([new Uint8Array([0xEF, 0xBB, 0xBF]), vcard], { type: "text/vcard;charset=utf-8" }); const url = window.URL.createObjectURL(blob); const a = document.createElement("a"); a.href = url; a.download = "김유진_연락처.vcf"; document.body.appendChild(a); a.click(); // 메모리 해제 setTimeout(() => { document.body.removeChild(a); window.URL.revokeObjectURL(url); alert("다운로드된 파일을 눌러 '연락처 저장'을 완료해 주세요!"); }, 500); }
